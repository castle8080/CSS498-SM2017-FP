let max2 a b = if a > b then a else b

let rec max xs =
  match xs with
    [x] -> x
    | hd :: tl -> (max2 hd (max tl));;

let rec max_aux current_max xs =
  match xs with
    [] -> current_max
    | x :: xss -> max_aux (max2 current_max x) xss

let maxx xs = max_aux (List.hd xs) (List.tl xs);;
