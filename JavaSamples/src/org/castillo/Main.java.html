package org.castillo;

import java.util.regex.Pattern;
import java.util.function.*;

class Greeter implements Function<String, String> {

    private final String[] parts;

    public Greeter(String format) {
        parts = format.split(Pattern.quote("#{name}"));
    }

    public String apply(String name) {
        return String.join(name, parts);
    }
}

public class Main {

    public static Function<String, String> createGreeter(String format) {
        String[] parts = format.split(Pattern.quote("#{name}"));
        return (String name) -> String.join(name, parts);
    }

    public static void main(String[] args) {
        Function<String, String> nice = createGreeter("Hi #{name}!");
        Function<String, String> mean = createGreeter("Go away #{name}!");

        System.out.println(nice.apply("Joe"));
        System.out.println(mean.apply("Bob"));
    }
}
